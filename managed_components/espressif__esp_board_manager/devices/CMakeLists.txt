# Function to get device sources
function(get_device_sources sources includes)
    set(srcs "")
    set(incs "")
    set(devices_dir "${CMAKE_CURRENT_LIST_DIR}/devices")

    if(CONFIG_ESP_BOARD_DEV_AUDIO_CODEC_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_audio_codec" "dev_audio_codec.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_DISPLAY_LCD_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_display_lcd" "dev_display_lcd.c")
        if(CONFIG_ESP_BOARD_DEV_DISPLAY_LCD_SUB_DSI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_display_lcd" "dev_display_lcd_sub_dsi.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_DISPLAY_LCD_SUB_SPI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_display_lcd" "dev_display_lcd_sub_spi.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_DISPLAY_LCD_SUB_PARLIO_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_display_lcd" "dev_display_lcd_sub_parlio.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_DISPLAY_LCD_SUB_RGB_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_display_lcd" "dev_display_lcd_sub_rgb.c")
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_GPIO_EXPANDER_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_gpio_expander" "dev_gpio_expander.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_LCD_TOUCH_I2C_SUPPORT AND NOT CONFIG_ESP_BOARD_DEV_LCD_TOUCH_SUB_I2C_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_lcd_touch_i2c" "dev_lcd_touch_i2c.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_LCD_TOUCH_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_lcd_touch" "dev_lcd_touch.c")
        if(CONFIG_ESP_BOARD_DEV_LCD_TOUCH_SUB_I2C_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_lcd_touch" "dev_lcd_touch_sub_i2c.c")
            list(APPEND incs "${devices_dir}/dev_lcd_touch_i2c")
            list(REMOVE_DUPLICATES incs)
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_GPIO_CTRL_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_gpio_ctrl" "dev_gpio_ctrl.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_FS_SPIFFS_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_fs_spiffs" "dev_fs_spiffs.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_LEDC_CTRL_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_ledc_ctrl" "dev_ledc_ctrl.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_LED_STRIP_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_led_strip" "dev_led_strip.c")
        if(CONFIG_ESP_BOARD_DEV_LED_STRIP_SUB_RMT_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_led_strip" "dev_led_strip_sub_rmt.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_LED_STRIP_SUB_SPI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_led_strip" "dev_led_strip_sub_spi.c")
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_CUSTOM_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_custom" "dev_custom.c")
    endif()

    if(CONFIG_ESP_BOARD_DEV_CAMERA_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_camera" "dev_camera.c")
        if(CONFIG_ESP_BOARD_DEV_CAMERA_SUB_CSI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_camera" "dev_camera_sub_csi.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_CAMERA_SUB_DVP_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_camera" "dev_camera_sub_dvp.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_CAMERA_SUB_SPI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_camera" "dev_camera_sub_spi.c")
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_POWER_CTRL_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_power_ctrl" "dev_power_ctrl.c")
        if(CONFIG_ESP_BOARD_DEV_POWER_CTRL_SUB_GPIO_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_power_ctrl" "dev_power_ctrl_sub_gpio.c")
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_FS_FAT_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_fs_fat" "dev_fs_fat.c")
        if(CONFIG_ESP_BOARD_DEV_FS_FAT_SUB_SDMMC_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_fs_fat" "dev_fs_fat_sub_sdmmc.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_FS_FAT_SUB_SPI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_fs_fat" "dev_fs_fat_sub_spi.c")
        endif()
    endif()

    if(CONFIG_ESP_BOARD_DEV_BUTTON_SUPPORT)
        bmgr_append_component_file(srcs incs "${devices_dir}" "dev_button" "dev_button.c")
        if(CONFIG_ESP_BOARD_DEV_BUTTON_SUB_GPIO_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_button" "dev_button_sub_gpio.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_BUTTON_SUB_ADC_SINGLE_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_button" "dev_button_sub_adc_single.c")
        endif()
        if(CONFIG_ESP_BOARD_DEV_BUTTON_SUB_ADC_MULTI_SUPPORT)
            bmgr_append_component_file(srcs incs "${devices_dir}" "dev_button" "dev_button_sub_adc_multi.c")
        endif()
    endif()

    # Set output variables in parent scope
    set(${sources} ${srcs} PARENT_SCOPE)
    set(${includes} ${incs} PARENT_SCOPE)
endfunction()
